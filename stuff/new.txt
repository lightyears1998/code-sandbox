#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;

void Reverse(char *, int);

int main()
{
    char s[105];
    
    /* 主要是输入处理的问题 */
    int i = 0;
    while (cin.get(s[i]))
    {
    	while (s[i] != '\n')
    	{
    		i++;
    		cin.get(s[i]);
    	}
        Reverse(s, i);
        s[i + 1] = '\0';
        cout << s;
        i = 0;
    }
    
    // while (fgets(s, 105, stdin))

}

void Reverse(char * s, int n)
{
    char c;
    for (int i = 0, j = n - 1; i < j; i++, j--)
    {
        c = s[i];
        s[i] = s[j];
        s[j] = c;
    }
}
